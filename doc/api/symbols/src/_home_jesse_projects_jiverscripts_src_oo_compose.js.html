  17 jive = this.jive || {};
  18 jive.oo = jive.oo || {};
  19 
  20 /**
  21  * Given two or more classes `compose()` creates a new class that inherits from
  22  * all of the composed classes.  The composed classes may have declare their
  23  * own superclasses; but they and their superclasses must form a linear
  24  * hierarchy.
  25  *
  26  * This implementation is similar to Traits as mixins in Scala, except that in
  27  * this implementation there is no distinction between classes and traits.
  28  *
  29  * @function
  30  * @param   {...jive.oo.Class}  classes     two or more classes to compose
  31  * @returns {jive.oo.Class} returns a new class that inherits from all of the composed classes
  32  * @requires jive.oo.Class
  33  */
  34 jive.oo.compose = (function(){
  35     // Returns true if the given array contains the given element and false
  36     // otherwise.
  37     function contains(array, e) {
  38         for (var i = 0; i < array.length; i += 1) {
  39             if (array[i] === e) {
  40                 return true;
  41             }
  42         }
  43         return false;
  44     }
  45 
  46     return function compose(/* classes */) {
  47         var classes = Array.prototype.slice.call(arguments)
  48           , original = jive.oo.Class
  49           , chain = []
  50           , ultimate = []
  51           , parent;
  52 
  53         for (var i = 0; i < classes.length; i += 1) {
  54             chain.push(classes[i]);  // add each class to the inheritance chain.
  55 
  56             // Record each class's parent class if it has one.
  57             parent = classes[i].superclass;
  58             if (parent !== original && !contains(ultimate, parent) && !contains(chain, parent)) {
  59                 ultimate.push(parent);
  60             }
  61         }
  62 
  63         // Only one ultimate class is allowed.
  64         if (ultimate.length > 1) {
  65             throw "Cannot create linear inheritance chain. Make sure classes that you are mixing in do not inherit from different parent classes.";
  66         } else if (ultimate.length == 1 && chain[0].superclass !== ultimate[0]) {
  67             chain.unshift(ultimate[0]);
  68         }
  69 
  70         var comp = chain[0];
  71         for (i = 1; i < chain.length; i += 1) {
  72             comp = comp.extend(chain[i].definition);
  73         }
  74 
  75         return comp;
  76     };
  77 })();
