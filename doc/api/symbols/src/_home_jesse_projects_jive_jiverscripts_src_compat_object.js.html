  17 /**
  18  * Returns a new object that has the given `obj` as its prototype.  This method
  19  * can be used to get prototypal inheritance without using the `new` keyword
  20  * directly.
  21  *
  22  * This implementation comes from:
  23  * http://ejohn.org/blog/ecmascript-5-objects-and-properties/
  24  *
  25  * @function
  26  * @param   {Object}    obj existing object to create a descendant of
  27  * @returns {Object}    a new object that inherits from `obj`
  28  */
  29 Object.create = Object.create || (function() {
  30     var defineProperty = Object.defineProperty || function(obj, name, desc) {
  31         if (desc.hasOwnProperty('value')) {
  32             obj[name] = desc.value;
  33         }
  34     };
  35 
  36     var defineProperties = Object.defineProperties || function(obj, props) {
  37         for (var name in props) {
  38             if (props.hasOwnProperty(name)) {
  39                 defineProperty(obj, name, props[name]);
  40             }
  41         }
  42     };
  43 
  44     return function( proto, props ) {
  45         var ctor = function( ps ) {
  46             if ( ps ) {
  47                 defineProperties( this, ps );
  48             }
  49         };
  50         ctor.prototype = proto;
  51         return new ctor( props );
  52     };
  53 })();

  55 /**
  56  * Returns an array of the attribute names on `obj`.
  57  *
  58  * @function
  59  * @param   {Object}    obj object to read keys from
  60  * @returns {string[]}  array of attribute names on `obj`
  61  */
  62 Object.keys = Object.keys || function(obj) {
  63     var keys = [];
  64     for ( var k in obj ) {
  65         if (obj.hasOwnProperty(k)) {
  66             keys.push(k);
  67         }
  68     }
  69     return keys;
  70 };
