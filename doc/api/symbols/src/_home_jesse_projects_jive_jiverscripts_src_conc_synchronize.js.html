  19 /**
  20  * A promise - an instance of jive.conc.Promise - is an object that represents
  21  * the eventual outcome of an asynchronous operation.  This function takes an
  22  * object that has promises as property values and returns a new promise.  Once
  23  * all of the promises on the given object are fulfilled `synchronize()` emits
  24  * success passing a copy of the original object with the success event with
  25  * any promise property values replaced with the outcomes of those promises.
  26  *
  27  * If any of the promises on the given object emits an error the same error
  28  * will be re-emitted by `synchronize()`.
  29  *
  30  * @function
  31  * @param   {Object|Array}  obj object referencing multiple promises
  32  * @returns {jive.conc.Promise} returns a promise that is fulfilled when all of the referenced promises have been fulfilled
  33  * @requires jive.conc.observable
  34  * @requires jive.conc.Promise
  35  */
  36 jive.conc.synchronize = (function() {
  37     function isArray(o) {
  38         return Object.prototype.toString.call(o) === "[object Array]";
  39     }
  40     
  41     function isPromise(prop) {
  42         return prop.addCallback && prop.addErrback && prop.addCancelback;
  43     }
  44 
  45     return function(obj) {
  46         var outcome, k,
  47             toBeFulfilled = 0,
  48             promise = new jive.conc.Promise();
  49     
  50         function addCallbacks(name, prop) {
  51             prop.addCallback(function(val) {
  52                 outcome[name] = val;
  53                 toBeFulfilled -= 1;
  54     
  55                 // Emit success when all promises are fulfilled.
  56                 if (toBeFulfilled < 1) {
  57                     promise.emitSuccess(outcome);
  58                 }
  59 
  60             }).addErrback(function(/* args */) {
  61                 promise.emitError.apply(promise, arguments);
  62 
  63             }).addCancelback(function(/* args */) {
  64                 promise.cancel.apply(promise, arguments);
  65             });
  66     
  67             toBeFulfilled += 1;
  68         }
  69     
  70         // Create a copy of the given object.  Handle synchronizing both arrays
  71         // and objects.
  72         outcome = isArray(obj) ? [] : {};
  73     
  74         for (k in obj) {
  75             if (obj.hasOwnProperty(k)) {
  76                 if (isPromise(obj[k])) {
  77                     addCallbacks(k, obj[k]);
  78                 } else {
  79                     outcome[k] = obj[k];
  80                 }
  81             }
  82         }
  83     
  84         return promise;
  85     };
  86 })();
